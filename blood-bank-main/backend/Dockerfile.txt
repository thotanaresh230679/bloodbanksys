# backend/Dockerfile
FROM openjdk:17-jdk-slim

# Install Maven
RUN apt-get update && apt-get install -y maven

# Set working directory
WORKDIR /app

# Copy the project files
COPY pom.xml .
COPY src ./src

# Build the application
RUN mvn clean package -DskipTests

# Expose the port
EXPOSE 8081

# Run the application
CMD ["java", "-jar", "target/blood-bank-backend-0.0.1-SNAPSHOT.jar"]
